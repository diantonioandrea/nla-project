% Bibliography.

\usepackage[backend=biber, style=authoryear, sorting=nyt]{biblatex}
\addbibresource{../bibliography/references.bib}
\setlength\bibhang{0pt}

\DeclareFieldFormat[article]{title}{\textit{#1}}
\DeclareFieldFormat[book]{title}{\textit{#1}}

\DeclareFieldFormat[article]{doi}{\href{https://doi.org/#1}{\textit{Article}}}
\DeclareFieldFormat[book]{doi}{\href{https://doi.org/#1}{\textit{Book}}}

\DeclareCiteCommand{\cite}
  {}
  {\printtext[bibhyperref]{\printnames{labelname} (\printfield{year})%
    \iffieldundef{postnote}{}{\addcomma\space\printfield{postnote}}}}
  {\multicitedelim}
  {}

\DeclareSortingTemplate{typealpha}{
  \sort{
    \field{entrytype} % Sort by type (book, article, etc.)
  }
  \sort{
    \field{author} % Then by author
  }
  \sort{
    \field{year} % Then by title
  }
}

\ExecuteBibliographyOptions{sorting=typealpha}

\DeclareBibliographyDriver{article} {%
    \noindent \textbf{\printnames{labelname}}
    (\printfield{doi}, \printfield{year})\par
    \printfield{title}\medskip
}

\DeclareBibliographyDriver{book} {%
    \noindent \textbf{\printnames{labelname}}
    (\printfield{doi}, \printfield{year})\par
    \printfield{title}\medskip
}
